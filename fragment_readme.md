## ğŸ’» AddCourseFragment Code Example

Here's how AddCourse would look as a **Fragment**:

```java
package com.example.coursemanager.fragments;

import android.os.Bundle;
import android.view.LayoutInflater;
import android.view.View;
import android.view.ViewGroup;
import android.widget.Button;
import android.widget.ProgressBar;
import android.widget.Toast;
import androidx.annotation.NonNull;
import androidx.annotation.Nullable;
import androidx.fragment.app.Fragment;
import com.example.coursemanager.R;
import com.example.coursemanager.models.Course;
import com.example.coursemanager.utils.FirebaseHelper;
import com.google.android.material.textfield.TextInputEditText;

public class AddCourseFragment extends Fragment {
    private TextInputEditText etCourseName, etCourseCode, etCreditHours;
    private Button btnAddCourse;
    private ProgressBar progressBar;
    private FirebaseHelper firebaseHelper;

    // Empty constructor required
    public AddCourseFragment() {
    }

    @Nullable
    @Override
    public View onCreateView(@NonNull LayoutInflater inflater, 
                             @Nullable ViewGroup container,
                             @Nullable Bundle savedInstanceState) {
        // Inflate layout for this fragment
        View view = inflater.inflate(R.layout.fragment_add_course, container, false);

        // Initialize views using view.findViewById (NOT just findViewById)
        etCourseName = view.findViewById(R.id.etCourseName);
        etCourseCode = view.findViewById(R.id.etCourseCode);
        etCreditHours = view.findViewById(R.id.etCreditHours);
        btnAddCourse = view.findViewById(R.id.btnAddCourse);
        progressBar = view.findViewById(R.id.progressBar);

        firebaseHelper = new FirebaseHelper();

        // Button click
        btnAddCourse.setOnClickListener(v -> addCourse());

        return view;
    }

    private void addCourse() {
        String name = etCourseName.getText().toString().trim();
        String code = etCourseCode.getText().toString().trim();
        String creditsStr = etCreditHours.getText().toString().trim();

        if (name.isEmpty() || code.isEmpty() || creditsStr.isEmpty()) {
            Toast.makeText(getContext(), "Please fill all fields", Toast.LENGTH_SHORT).show();
            return;
        }

        int credits = Integer.parseInt(creditsStr);
        progressBar.setVisibility(View.VISIBLE);
        btnAddCourse.setEnabled(false);

        Course course = new Course(null, name, code, credits);

        firebaseHelper.addCourse(course,
            unused -> {
                progressBar.setVisibility(View.GONE);
                Toast.makeText(getContext(), "Course added!", Toast.LENGTH_SHORT).show();
                
                // Go back to previous fragment
                getParentFragmentManager().popBackStack();
            },
            e -> {
                progressBar.setVisibility(View.GONE);
                btnAddCourse.setEnabled(true);
                Toast.makeText(getContext(), "Failed: " + e.getMessage(), 
                        Toast.LENGTH_SHORT).show();
            }
        );
    }
}
```

### Key Differences from Activity:

| Activity | Fragment |
|----------|----------|
| `extends AppCompatActivity` | `extends Fragment` |
| `setContentView(R.layout.xxx)` | `inflater.inflate(R.layout.xxx, container, false)` |
| `findViewById(R.id.xxx)` | `view.findViewById(R.id.xxx)` |
| `this` for context | `getContext()` |
| `finish()` to go back | `getParentFragmentManager().popBackStack()` |
| Uses `Intent` to open | Uses `FragmentTransaction` to show |

### From CoursesFragment, open AddCourseFragment:
```java
// In CoursesFragment, when FAB is clicked:
fabAddCourse.setOnClickListener(v -> {

    AddCourseFragment addFragment = new AddCourseFragment();
    
    // Replace current fragment with new one
    getParentFragmentManager()
        .beginTransaction()

        .addToBackStack(null)  // Allows back button to work
        .commit();
```
